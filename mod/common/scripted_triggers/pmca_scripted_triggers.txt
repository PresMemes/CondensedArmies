# ID Trigger
has_condensed_armies = {
    always = yes
}

# Hides the given army if a given flag is not set
pmca_flag_check = {
    # optimize_memory
    OR = {
        is_ai = no
        has_global_flag = pmca_ai_allowed
    }
    has_country_flag = pmca_times_$MULT$_set
}

# Check if the army is a Condensed Army
pmca_is_condensed_army = {
    optimize_memory
    OR = {
        has_army_flag = pmca_times_ten_army
        has_army_flag = pmca_times_hundred_army
        pmca_is_times_ten_army = yes
        pmca_is_times_hundred_army = yes
    }
}

# Check if the army is a x10 Condensed Army
pmca_is_times_ten_army = {
    optimize_memory
    OR = {
        has_army_flag = pmca_times_ten_army

        # Vanilla x10
        inline_script = {
            script = compatibility/pmca_conditional_inline_1_input

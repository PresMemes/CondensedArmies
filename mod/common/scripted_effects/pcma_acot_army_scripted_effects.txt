# Creates an army based on an ACOT tech
# Scopes: this = country, fromfromform = planet
# Parameters: MULT = ten/hundred
pmca_acot_army_create_infantry_crew_survivor = {
	inline_script = {
		script = army/pmca_conditional_inline_p1
		MOD_VARIABLE = @has_ancient_caches
		CONTENT = "
		switch = {
			trigger = has_technology
			tech_acot_army_se = {
				fromfromfrom = {
					pmca_spawn_condensed_army = {
						TYPE = pmca_$ARMY_MULT$_acot_infantry_se
						MULT_FLAG = $ARMY_MULT$
					}
				}
			}
			tech_acot_army_ae = {
				fromfromfrom = {
					pmca_spawn_condensed_army = {
						TYPE = pmca_$ARMY_MULT$_acot_infantry_ae
						MULT_FLAG = $ARMY_MULT$
					}
				}
			}
			tech_acot_army_dm = {
				fromfromfrom = {
					pmca_spawn_condensed_army = {
						TYPE = pmca_$ARMY_MULT$_acot_infantry_dm
						MULT_FLAG = $ARMY_MULT$
					}
				}
			}
			default = {
				fromfromfrom = {
					pmca_spawn_condensed_army = {
						TYPE = pmca_$ARMY_MULT$_assault_army
						MULT_FLAG = $ARMY_MULT$
					}
				}
			}
		}
		"
	}
}
